备战蓝桥杯笔记（七）—二分（二）

2022-01-19

一个简单的例题

分巧克力

👍2017年蓝桥杯真题👍
原题链接
题意：
有N块巧克力需要分给K位小朋友，第 i 块是 Hi×Wi 的方格组成的长方形
分出的巧克力需要满足以下条件
1.形状是正方形，边长是整数
2.大小相同

样例

输入：

2 10
6 5
5 6

输出：
2

思路
根据题目意思可知，求满足条件区间右端点，直接使用二分模板
然后完善一下check函数即可

代码

#include<bits/stdc++.h>
using namespace std;
int n,m,k,i,sum,h[100005],w[100005];
int max(int x,int y,int z)//三个数之间求最大值函数 
{
	int e,f;
	
	e=x>y?x:y;
	
	f=e>z?e:z;
	
	return f; 

}
int check(int x)
{
	int sum=0;
	for(int j=0;j<n;j++)
	{
		sum+=(h[j]/x)*(w[j]/x);//计算每块巧克力可以被分出多少 
	}
	if(sum<k)
	return 0;
	else return 1;
}
int main()
{
	
	while(scanf("%d %d",&n,&k)!=EOF)
	{
		for(int i=0;i<n;i++)
			{
				scanf("%d %d",&h[i],&w[i]);
				m=max(m,h[i],w[i]);//求出数据最大的范围 
			}
		int l=0,r=m;//二分模板 
		while(l<r)
		{
			int mid=(l+r+1)/2;
			if(check(mid))
				l=mid;
				else
				r=mid-1;
				
		}
		printf("%d\n",l);
	}
 }
